-- server
keytool -genkeypair -keyalg RSA -keysize 2048   -validity 365 -storetype PKCS12   -keystore server.p12 -storepass serverpass   -alias server -dname "CN=NettyServer"   -ext SAN=DNS:localhost,IP:127.0.0.1

keytool -exportcert -keystore server.p12 -storepass serverpass   -alias server -file server.cer


-- client
keytool -importcert -file server.cer -alias server   -keystore client-truststore.p12 -storepass clientpass   -storetype PKCS12 -noprompt
